write-host -ForegroundColor Yellow "List of Groups with RequireSenderAuthenticationEnabled set to False:"

get-distributiongroup -ResultSize Unlimited -IgnoreDefaultScope | where-object {$_.RequireSenderAuthenticationEnabled -like 'false' } | fl alias,RequireSenderAuthenticationEnabled

write-host -ForegroundColor Red "Values from Spreadsheet:"
Import-Csv C:\distributiongroups_modified.csv | where-object {$_.RequireSenderAuthenticationEnabled -like 'false' } | ForEach-Object {
write-host -ForegroundColor Yellow $_.alias $_.RequireSenderAuthenticationEnabled
set-distributiongroup $_.alias -RequireSenderAuthenticationEnabled $false
}
write-host -ForegroundColor Red "Checking groups in Dev AD..."
write-host -ForegroundColor Red "Values In Dev AD:"
get-distributiongroup | where-object {$_.RequireSenderAuthenticationEnabled -like 'false' } | ForEach-Object {
write-host -ForegroundColor Green $_.alias $_.RequireSenderAuthenticationEnabled
}
